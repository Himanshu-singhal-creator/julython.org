{% trans "How do I add a web hook for my project?" %}

{% trans "Github" %}

{% trans "If you host your project on github, follow the instructions below to setup a post receive hook." %}

  1. {% trans "Go to your project on github." %}
  2. {% trans "Select the 'Settings' link on the project dashboard." %}
  3. {% trans "Select 'Webhooks and Services' on the left menu." %}
  4. {% trans "Select 'Add webhook'." %}
  5. {% trans "Enter the Payload URL: 
    http://www.julython.org/api/v1/github
    " %}
  6. {% trans "Choose Content type: 
    application/x-www-form-urlencoded
    " %}
  7. {% trans "There is no secret, and we just want the 'push' event sent" %}
  8. {% trans "Make sure it is active and select 'Add Webhook'" %}

{% trans "You can validate and re-test the hook by selecting it expanding the 'Recent Deliveries' and select 'Redeliver'" %}

{% trans "Bitbucket" %}

{% trans "If you host your project on bitbucket, follow the instructions below to setup a post receive hook." %}

  1. {% trans "Go to your project on bitbucket." %}
  2. {% trans "Select the 'Admin' admin tab on your project dashboard." %}
  3. {% trans "Select 'Services' on the left menu." %}
  4. {% trans "In the 'Select a service' dropdown select or type 'POST' then 'Add service'." %}
  5. {% trans "Enter the URL: 
    http://www.julython.org/api/v1/bitbucket
    " %}
  6. {% trans "Select 'Save Settings'" %}

{% trans "You can test it out by pushing a commit to your repository." %}
